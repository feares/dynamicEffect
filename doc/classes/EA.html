    • stop
    <html>
    <body>
        <style>
            .container {
               width: 500px;
               margin: 200px auto;
            }
            #demo-div {
                width: 100px;
                height: 100px;
                margin-left: 200px;
                background: #b8f1d2;
            }
            .ea-rule {
                -webkit-animation: 2s ease;
                   -moz-animation: 2s ease;
                    -ms-animation: 2s ease;
                     -o-animation: 2s ease;
                        animation: 2s ease;
            }
            .ea-flash {
                    -webkit-animation-name: ea-flash;
                    -moz-animation-name: ea-flash;
                    -ms-animation-name: ea-flash;
                    -o-animation-name: ea-flash;
                    animation-name: ea-flash;
            }

            @-webkit-keyframes ea-flash {
                    0%, 50%, 100% {
                            opacity: 1;
                    }
                    25%, 75% {
                            opacity: 0;
                    }
            }

            @-moz-keyframes ea-flash {
                    0%, 50%, 100% {
                            opacity: 1;
                    }
                    25%, 75% {
                            opacity: 0;
                    }
            }

            @-ms-keyframes ea-flash {
                    0%, 50%, 100% {
                            opacity: 1;
                    }
                    25%, 75% {
                            opacity: 0;
                    }
            }

            @-o-keyframes ea-flash {
                    0%, 50%, 100% {
                            opacity: 1;
                    }
                    25%, 75% {
                            opacity: 0;
                    }
            }

            @keyframes ea-flash {
                    0%, 50%, 100% {
                            opacity: 1;
                    }
                    25%, 75% {
                            opacity: 0;
                    }
            }
        </style>
        <div class="container">
            <div id="demo-div" class="ea-rule"></div>
        </div>
    </body>
    <script>
        require.config({
            baseUrl: './res'
        });
        require(['ea'], function (ea) {
            ea('#demo-div').trigger('ea-flash', true);
            setTimeout(function () {
                ea('#demo-div').stop('ea-flash');
            }, 1000);
        });
    </script>
</html>
