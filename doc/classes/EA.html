        require.config({
            baseUrl: './res'
        });
        require(['ea'], function (ea) {
            var name1 = ea('#id');
            console.log(name1);
            var name2 = ea('.className');
            console.log(name2);
            var name3 = ea(document.getElementById('div3'));
            console.log(name3);
            var name4 = ea($('#div4'));
            console.log(name4);
        });
    </script>
